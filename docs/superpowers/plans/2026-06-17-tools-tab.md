# Tools Tab Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a third `Tools` tab to DropComp with one-click After Effects layer utilities (anchor grid, layer creators, align/distribute + sequence, pre-comp tools), so standalone tool panels can be undocked.

**Architecture:** A new custom (non-card) tab. The panel's tab switcher is generalized so a tab is either a *card view* (Library/Assets) or a *custom view* (Tools). Pure, runtime-used logic (anchor fractions, Num/Step clamping, tab coercion) lives in node-tested panel modules; AE operations live in a new ES3 host module `jsx/tools.jsx` loaded through the existing `loadHostModules` mechanism. The panel sends primitive args to the host; every host op is one undo group, guarded, and serialized through the existing `DCBridge` lock.

**Tech Stack:** Adobe CEP. Panel JS = Chromium 99 (modern JS, IIFE modules exposing a `DCx` global + optional `module.exports`). Host = ExtendScript ES3 only. Tests = `node --test` (zero deps).

## Global Constraints

- Panel JS may use modern JS; `jsx/*.jsx` is **ES3 only** — no `const`/`let`, no arrow functions, no template literals, no `.map/.filter/.forEach/.reduce/.some/.every`, no `Object.keys`, no `Array.isArray` (statically enforced by `tests/jsx.es3.test.js`, which globs every `jsx/*.jsx`).
- Every top-level function in a loaded jsx module **must** export to `$.global` (`$.global.fn = fn;`) — `$.evalFile` runs in local scope (enforced by `tests/jsx.exports.test.js`).
- New AE operations go in `jsx/tools.jsx`, **never** in `jsx/hostscript.jsx` (already 755 lines with a standing split-TODO).
- Version floor: AE 15.0+ (`CSXS/manifest.xml`). `tlDecompose` additionally requires AE 2022+ and degrades gracefully below it.
- File-size contract: target < 400 lines/file; `jsx/*.jsx` hard limit 800.
- Release coupling: `package.json`, `CSXS/manifest.xml`, and `panel/js/update.js` versions must agree or `tests/update.test.js` fails.
- Every host op: wrap in one `app.beginUndoGroup(...)`/`endUndoGroup()`; guard preconditions; return JSON `{"ok":true,...}` or `jerr('message')`. Never throw to the panel.
- Theme tokens (CSS vars in `panel/css/style.css`): `--bg #161616`, `--bg-raised #222`, `--bg-inset #1d1d1d`, `--bg-hover #2a2a2a`, `--border #2e2e2e`, `--text #f0f0f0`, `--text-dim #777`, `--gold #ffd700`, `--radius 8px`.

```javascript
var DCToolsCore = (function () {
  'use strict';

  // Row-major 3x3 grid index (0=top-left .. 8=bottom-right) -> bounding-box
  // fractions. Used at runtime by the panel and sent to the host as primitives.
  function anchorFraction(index) {
    var i = (typeof index === 'number') ? index : parseInt(index, 10);
    if (isNaN(i) || i < 0 || i > 8) i = 4;
    return [(i % 3) * 0.5, Math.floor(i / 3) * 0.5];
  }

  function clampInt(value, min, max, fallback) {
    var n = parseInt(value, 10);
    if (isNaN(n)) return fallback;
    if (n < min) return min;
    if (n > max) return max;
    return n;
  }

  return { anchorFraction: anchorFraction, clampInt: clampInt };
}());
if (typeof module !== 'undefined' && module.exports) { module.exports = DCToolsCore; }
```

Create `panel/js/tools.js`:

```javascript
var DCTools = (function () {
  'use strict';

  var mounted = false;
  var PRECOMP_FN = {
    precomp: 'tlPreComp', multi: 'tlMultiPreComp',
    decompose: 'tlDecompose', independent: 'tlIndependent'
  };
  var OK_MSG = {
    tlSetAnchor: 'Anchor set.', tlCreateLayer: 'Layer created.',
    tlAlign: 'Aligned.', tlDistribute: 'Distributed.', tlReset: 'Recentered.',
    tlSequence: 'Sequenced.', tlPreComp: 'Precomposed.',
    tlMultiPreComp: 'Precomposed each layer.', tlDecompose: 'Decomposed.',
    tlIndependent: 'Made independent.'
  };

  function init() { /* nothing to pre-load; wiring is lazy in ensureMounted */ }

  function ensureMounted() {
    if (mounted) return;
    mounted = true;
    var root = document.getElementById('tools');
    root.addEventListener('click', onClick);
  }

  function run(label, fn, args) {
    if (!DCBridge.acquire(label)) { DCUI.toast('Busy: ' + DCBridge.busyWith(), true); return; }
    DCBridge.call(fn, args, function (result) {
      DCBridge.release();
      var r = DCBridge.parseJson(result);
      if (r && r.ok) DCUI.toast(OK_MSG[fn] || 'Done.', false);
      else DCUI.toast((r && r.error) || result || 'Tool failed.', true);
    });
  }

  function onClick(e) {
    var btn = e.target.closest('[data-tool]');
    if (!btn) return;
    var tool = btn.dataset.tool;
    var arg = btn.dataset.arg;
    if (tool === 'anchor') {
      var f = DCToolsCore.anchorFraction(parseInt(arg, 10));
      run('anchor', 'tlSetAnchor', [f[0], f[1]]);
    } else if (tool === 'create') {
      run('create', 'tlCreateLayer', [arg]);
    } else if (tool === 'align') {
      run('align', 'tlAlign', [arg]);
    } else if (tool === 'distribute') {
      run('distribute', 'tlDistribute', [arg]);
    } else if (tool === 'reset') {
      run('reset', 'tlReset', []);
    } else if (tool === 'sequence') {
      var num = DCToolsCore.clampInt(document.getElementById('tools-num').value, 1, 500, 1);
      var step = DCToolsCore.clampInt(document.getElementById('tools-step').value, -100000, 100000, 5);
      run('sequence', 'tlSequence', [num, step]);
    } else if (tool === 'precomp') {
      run('precomp', PRECOMP_FN[arg], []);
    }
  }

  return { init: init, ensureMounted: ensureMounted };
}());
```

```javascript
// DropComp Tools host module (ExtendScript, ES3 only). Loaded by
// loadHostModules in hostscript.jsx; every function exports to $.global.
// Reuses jerr() and jsonEscape() defined in hostscript.jsx (host-global).

function tlActiveComp() {
    var c = app.project ? app.project.activeItem : null;
    return (c && c instanceof CompItem) ? c : null;
}
$.global.tlActiveComp = tlActiveComp;

function tlSelectOnly(comp, layer) {
    for (var i = 1; i <= comp.numLayers; i++) {
        comp.layer(i).selected = (comp.layer(i) === layer);
```

```javascript
function tlLayerBounds(comp, layer) {
    var rect = layer.sourceRectAtTime(comp.time, false);
    var t = layer.property('ADBE Transform Group');
    var pos = t.property('ADBE Position').value;
    var anc = t.property('ADBE Anchor Point').value;
```

```javascript
function tlDecompose() {
    var comp = tlActiveComp();

    var sel = comp.selectedLayers;
```
